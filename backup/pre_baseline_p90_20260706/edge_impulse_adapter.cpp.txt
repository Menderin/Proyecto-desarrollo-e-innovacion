#include "edge_impulse_adapter.h"

#include <cstdlib>
#include <cstdint>

#include "detector_thresholds.h"

esp_err_t edge_impulse_adapter_init(void)
{
    return ESP_OK;
}

classifier_result_t edge_impulse_adapter_run(const acquisition_buffer_t *buffer)
{
    if (buffer == nullptr || buffer->frame_count == 0) {
        return CLASSIFIER_RESULT_EMPTY;
    }

    int32_t max_delta_sq = 0;
    bool has_reference = false;
    magnetometer_axes_t reference[SENSOR_COUNT] = {};

    for (size_t frame_idx = 0; frame_idx < buffer->frame_count; frame_idx++) {
        const acquisition_frame_t *frame = &buffer->frames[frame_idx];

        for (size_t sensor_idx = 0; sensor_idx < SENSOR_COUNT; sensor_idx++) {
            const magnetometer_axes_t *axes = &frame->sensors[sensor_idx].axes;

            if (!has_reference) {
                reference[sensor_idx] = *axes;
                continue;
            }

            int32_t dx = (int32_t)axes->x - reference[sensor_idx].x;
            int32_t dy = (int32_t)axes->y - reference[sensor_idx].y;
            int32_t dz = (int32_t)axes->z - reference[sensor_idx].z;
            int32_t delta_sq = (dx * dx) + (dy * dy) + (dz * dz);

            if (delta_sq > max_delta_sq) {
                max_delta_sq = delta_sq;
            }
        }

        has_reference = true;
    }

    int32_t threshold = DETECTOR_MAG_THRESHOLD_RAW;
    if (threshold < 180) {
        threshold = 180;
    }

    int32_t threshold_sq = threshold * threshold;
    return max_delta_sq >= threshold_sq ? CLASSIFIER_RESULT_ALERT : CLASSIFIER_RESULT_SAFE;
}
